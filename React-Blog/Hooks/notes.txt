Popular React Hooks:
--------------------
. useState
. useEffect
. useRef
. useContext
. useReducer

useEffect(()=>(
    // call every time
))

useEffect(()=>(
    // call only one time
),[])

useEffect(()=>(
    // call on changing the respective state
),[state])

useEffect(()=>(
    // call on changing both the state
),[state1] [state2])

useEffect(()=>(
    // call on changing the props
),[prop1, prop2])
